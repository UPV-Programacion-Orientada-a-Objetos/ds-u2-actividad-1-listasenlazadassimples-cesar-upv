156void recortarEspacios(char* texto)
157{
158 if (!texto)
159 {
160 return;
161 }
162
163 std::size_t inicio = 0;
164 while (texto[inicio] == ' ' || texto[inicio] == '\t')
165 {
166 ++inicio;
167 }
168
169 if (inicio > 0)
170 {
171 std::size_t i = 0;
172 while (texto[inicio + i] != '\0')
173 {
174 texto[i] = texto[inicio + i];
175 ++i;
176 }
177 texto[i] = '\0';
178 }
179
180 int fin = static_cast<int>(std::strlen(texto)) - 1;
181 while (fin >= 0 && (texto[fin] == ' ' || texto[fin] == '\t'))
182 {
183 texto[fin] = '\0';
184 --fin;
185 }
186}

191bool descomponerLineaSerial(const char* linea, char* id, std::size_t tamId, char* valor, std::size_t tamValor)
192{
193 if (!linea || !id || !valor)
194 {
195 return false;
196 }
197
198 std::size_t i = 0;
199 std::size_t j = 0;
200 while (linea[i] != '\0' && linea[i] != ',' && j < tamId - 1)
201 {
202 id[j++] = linea[i++];
203 }
204 id[j] = '\0';
205
206 if (linea[i] != ',')
207 {
208 return false;
209 }
210 ++i;
211
212 std::size_t k = 0;
213 while (linea[i] != '\0' && k < tamValor - 1)
214 {
215 valor[k++] = linea[i++];
216 }
217 valor[k] = '\0';
218
219 recortarEspacios(id);
220 recortarEspacios(valor);
221
222 return (id[0] != '\0' && valor[0] != '\0');
223}

257bool configurarPuertoSerial(int fd, AuxiliarCli& cli)
258{
259 struct termios opciones;
260 if (tcgetattr(fd, &opciones) != 0)
261 {
262 cli.imprimirLog("WARNING", "No se pudo obtener la configuración del puerto serial.");
263 return false;
264 }
265
266 cfsetispeed(&opciones, B9600);
267 cfsetospeed(&opciones, B9600);
268
269 opciones.c_cflag = CS8 | CLOCAL | CREAD;
270 opciones.c_iflag = IGNPAR;
271 opciones.c_oflag = 0;
272 opciones.c_lflag = 0;
273 opciones.c_cc[VMIN] = 0;
274 opciones.c_cc[VTIME] = 10;
275
276 if (tcsetattr(fd, TCSANOW, &opciones) != 0)
277 {
278 cli.imprimirLog("WARNING", "No se pudo establecer la configuración del puerto serial.");
279 return false;
280 }
281
282 tcflush(fd, TCIOFLUSH);
283 return true;

313

317 FD_ZERO(&conjuntoLectura);
318 FD_SET(fd, &conjuntoLectura);

320 int maximo = (fd > STDIN_FILENO) ? fd : STDIN_FILENO;
321
322 int resultado = select(maximo + 1, &conjuntoLectura, nullptr, nullptr, nullptr);
323 if (resultado < 0)
324 {
325 if (errno == EINTR)
326 {
327 continue;
328 }
329 cli.imprimirLog("WARNING", "Select interrumpido, se detiene la lectura serial.");
330 break;
331 }
332
333 if (FD_ISSET(STDIN_FILENO, &conjuntoLectura))
334 {
335 char descarte[8];
336 if (fgets(descarte, sizeof(descarte), stdin) != nullptr)
337 {
338 cli.imprimirLog("STATUS", "Lectura serial detenida por el usuario.");
339 leyendo = false;
340 }
341 }
342
343 if (!leyendo)
344 {
345 break;
346 }
347
348 if (FD_ISSET(fd, &conjuntoLectura))
349 {
350 char byteLeido = 0;
351 ssize_t cantidad = read(fd, &byteLeido, 1);
352 if (cantidad > 0)
353 {
354 if (byteLeido == '\r')
355 {
356 continue;
357 }
358
359 if (byteLeido == '\n')
360 {
361 if (posicion > 0)
362 {
363 linea[posicion] = '\0';
364 char id[TAM_ID] = {0};
365 char valorCadena[40] = {0};
366
367 if (descomponerLineaSerial(linea, id, TAM_ID, valorCadena, sizeof(valorCadena)))
368 {
369 SensorBase* sensor = lista.buscarPorNombre(id);
370 if (sensor)

372 sensor->registrarLecturaDesdeCadena(valorCadena);
373 }
374 else
375 {
376 char mensaje[160];
377 std::snprintf(mensaje, sizeof(mensaje), "Sensor '%s' no se encuentra en la lista.", id);
378 cli.imprimirLog("WARNING", mensaje);
379 }
380 }
381 else
382 {
383 cli.imprimirLog("WARNING", "Lectura serial ignorada: formato incorrecto.");
384 }
385 }
386 posicion = 0;
387 }
388 else
389 {
390 if (posicion < sizeof(linea) - 1)
391 {
392 linea[posicion++] = byteLeido;
393 }
394 }
395 }
396 else if (cantidad == 0)
397 {
398 cli.imprimirLog("STATUS", "No se reciben datos; posible desconexión del dispositivo.");
399 leyendo = false;
400 }
401 else if (errno != EAGAIN && errno != EWOULDBLOCK)
402 {
403 cli.imprimirLog("WARNING", "Lectura serial interrumpida por un error.");
404 leyendo = false;
405 }
406 }
407 }
